ENTRY(primary_cpu_entry)

SECTIONS {
    . = 0x80000;  // Load address

    .text : {
        *(.text.boot)
        *(.text)
    }

    .rodata : {
        *(.rodata)
    }

    .data : {
        *(.data)
    }

    .bss : {
        __bss_start = .;
        *(.bss)
        __bss_end = .;
    }

    /* Stack top for kernel */
    __kernel_stack_top = 0x100000;

    /* Kernel end */
    . = ALIGN(4096);
}
